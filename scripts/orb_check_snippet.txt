  # ─── OrbStack hostname check ──────────────────────────────────
  orb-check:
    desc: Verify OrbStack hostnames are reachable (use these in Cursor)
    cmds:
      - |
        echo "Checking OrbStack hostnames..."
        echo ""

        # LiteLLM
        if curl -sf --max-time 3 {{.LITELLM_ORB}}/health/liveliness >/dev/null 2>&1; then
          echo "LiteLLM OK  -> {{.LITELLM_ORB}}/v1  (use this in Cursor)"
        else
          echo "LiteLLM FAIL -> stack may not be running: task up"
        fi

        echo ""
        echo "Router container status:"
        docker inspect ollama-router --format "  name={{.Name}} hostname={{.Config.Hostname}} status={{.State.Status}}" 2>/dev/null || echo "  container not found"

        echo ""
        echo "Trying router hostname variants..."
        for host in \
          "ollama-router.ollama-stack.orb.local" \
          "router.ollama-stack.orb.local" \
          "ollama-router.orb.local"; do
          if curl -sf --max-time 2 "http://${host}:4001/health" >/dev/null 2>&1; then
            echo "  REACHABLE -> http://${host}:4001"
          else
            echo "  unreachable -> http://${host}:4001"
          fi
        done

        echo ""
        echo "Localhost fallback (works for task router-test, not Cursor):"
        if curl -sf --max-time 2 "http://localhost:4001/health" >/dev/null 2>&1; then
          echo "  OK -> http://localhost:4001"
        else
          echo "  FAIL -> router not running or unhealthy"
          echo "  Try: task restart -- router && sleep 15 && task orb-check"
        fi
