  # ─── OrbStack hostname check ──────────────────────────────────
  orb-check:
    desc: Verify OrbStack hostnames and container status
    cmds:
      - |
        echo "=== LiteLLM ==="
        if curl -sf --max-time 3 {{.LITELLM_ORB}}/health/liveliness >/dev/null 2>&1; then
          echo "OK  -> {{.LITELLM_ORB}}/v1  (configured in Cursor)"
        else
          echo "FAIL -> run: task up"
        fi

        echo ""
        echo "=== Router container ==="
        docker inspect ollama-router --format "  status={{.State.Status}} hostname={{.Config.Hostname}}" 2>/dev/null || echo "  not found"

        echo ""
        echo "=== Router hostname variants ==="
        for host in \
          "ollama-router.ollama-stack.orb.local" \
          "router.ollama-stack.orb.local" \
          "ollama-router.orb.local"; do
          if curl -sf --max-time 2 "http://${host}:4001/health" >/dev/null 2>&1; then
            echo "  OK  -> http://${host}:4001"
          else
            echo "  --     http://${host}:4001"
          fi
        done

        echo ""
        echo "=== Localhost fallback ==="
        if curl -sf --max-time 2 "http://localhost:4001/health" >/dev/null 2>&1; then
          echo "  OK  -> http://localhost:4001  (for task router-test)"
        else
          echo "  FAIL -> try: task restart -- router && sleep 20 && task orb-check"
        fi
